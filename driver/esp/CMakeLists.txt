file(GLOB ${PROJECT_NAME}_DRIVER_SOURCES CONFIGURE_DEPENDS
     "${CMAKE_CURRENT_LIST_DIR}/*.cpp"
)
list(SORT ${PROJECT_NAME}_DRIVER_SOURCES)

if(DEFINED CONFIG_IDF_TARGET_ESP32P4 AND CONFIG_IDF_TARGET_ESP32P4)
  list(REMOVE_ITEM ${PROJECT_NAME}_DRIVER_SOURCES
       "${CMAKE_CURRENT_LIST_DIR}/esp_wifi_client.cpp"
       "${CMAKE_CURRENT_LIST_DIR}/esp_uart.cpp"
       "${CMAKE_CURRENT_LIST_DIR}/esp_uart_dma.cpp"
       "${CMAKE_CURRENT_LIST_DIR}/esp_uart_fifo.cpp")
elseif(DEFINED CONFIG_SOC_WIFI_SUPPORTED AND NOT CONFIG_SOC_WIFI_SUPPORTED)
  list(REMOVE_ITEM ${PROJECT_NAME}_DRIVER_SOURCES
       "${CMAKE_CURRENT_LIST_DIR}/esp_wifi_client.cpp")
endif()

target_sources(${PROJECT_NAME} PRIVATE ${${PROJECT_NAME}_DRIVER_SOURCES})
